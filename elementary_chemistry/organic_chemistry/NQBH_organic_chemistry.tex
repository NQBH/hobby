\subsection{Hợp Chất Hữu Cơ \& Hóa Học Hữu Cơ}

\begin{dinhnghia}[Hợp chất\texttt{/}Hóa học hữu cơ]
	\emph{Hợp chất hữu cơ} là hợp chất của carbon (trừ \emph{\ce{CO,CO2}}, muối carbonat, xianua, cacbua, $\ldots$). \emph{Hóa học hữu cơ} là ngành hóa học chuyên nghiên cứu các hợp chất hữu cơ.
\end{dinhnghia}
